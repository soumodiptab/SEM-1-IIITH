#include <linux/syscalls.h>
#include <linux/kernel.h>
#include <linux/cred.h>
SYSCALL_DEFINE1(soumodiptaprint, char __user *,data)
{
	char buffer[256];
	long copied = strncpy_from_user(buffer, data, sizeof(buffer));
  	if (copied < 0 || copied == sizeof(buffer))
    		return -EFAULT;
	printk("%s\n",buffer);
	printk("%s\n",data);
    return 1;
}
